name: Build & Release NMG Analyse

on:
  push:
    tags:
      - 'v*'
  workflow_dispatch:
    inputs:
      version:
        description: 'Version (z.B. 1.5.0)'
        required: true

jobs:
  build:
    runs-on: windows-latest

    steps:
      - name: Checkout Code
        uses: actions/checkout@v4

      - name: Python einrichten
        uses: actions/setup-python@v5
        with:
          python-version: '3.11'

      - name: Abhängigkeiten installieren
        run: |
          pip install pyinstaller openpyxl requests

      - name: Version ermitteln
        id: version
        run: |
          $version = "${{ github.event.inputs.version }}"
          if (-not $version) {
            $version = "${{ github.ref_name }}" -replace "^v", ""
          }
          echo "VERSION=$version" >> $env:GITHUB_OUTPUT
        shell: pwsh

      - name: Build .exe
        run: |
          python build_release.py --version ${{ steps.version.outputs.VERSION }} --build-only

      - name: Release erstellen und hochladen
        uses: softprops/action-gh-release@v2
        with:
          tag_name: v${{ steps.version.outputs.VERSION }}
          name: NMG Analyse v${{ steps.version.outputs.VERSION }}
          files: |
            releases/*.zip
          generate_release_notes: true
        env:
          GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}

      - name: release.json aktualisieren
        run: |
          $info = @{
            version = "${{ steps.version.outputs.VERSION }}"
            name = "NMG Analyse v${{ steps.version.outputs.VERSION }}"
            date = (Get-Date).ToString("yyyy-MM-dd")
            download_url = "https://github.com/${{ github.repository }}/releases/download/v${{ steps.version.outputs.VERSION }}/NMG_Analyse_v${{ steps.version.outputs.VERSION }}.zip"
          } | ConvertTo-Json
          $info | Out-File -FilePath release.json -Encoding utf8
          git config user.name "github-actions[bot]"
          git config user.email "github-actions[bot]@users.noreply.github.com"
          git add release.json
          git commit -m "Update release.json to v${{ steps.version.outputs.VERSION }}" || echo "Keine Änderungen"
          git push
        shell: pwsh
        env:
          GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}
